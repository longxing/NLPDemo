本地NLU能力支持的语法示例
系统词典：

{
"tag_hi": {
"lexicon": [
"你好",
"hello",
"hi"
]
},
"tag_hi_ext": {
"lexicon": [
"[tag_hi]",
"早上好",
"晚上好"
]
},
"tag_city": {
"lexicon": [
"北京",
"上海",
"广州"
]
}
}

用户词典：

{
"tag_pay": {
"lexicon": [
"支付",
"付款",
"付钱"
]
},
"tag_currency": {
"lexicon": [
"美元",
"人民币",
"欧元",
"日元"
]
}
}

用户语法：

{
"pat_hi": {
"phrase": [
"[tag_hi]|[tag_hi_ext]"
]
},
"pat_location": {
"phrase": [
"(我在|我住)[tag_city]"
]
},
"pat_want": {
"phrase": [
"我想|我要"
]
},
"pat_pay": {
"phrase": [
"(([tag_pay][tag_currency]))"
]
}
}

用户意图：

{
"ent_pay_money": {
"condition": [
"HIT(pat_hi) and HIT(pat_location) and HIT(pat_want) and (HIT(pat_pay) or true) and CVR(40)"
],
"variable": [
"(B::HIT::pat_pay)city=S::VAR::pat_location::tag_city",
"currency=S::VAR::pat_pay::tag_currency"
]
}
}

以上语法资源均可通过设置适用不同场景，更多语法设置方式请咨询捷通华声或关注开发者社区
